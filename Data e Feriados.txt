Sub Trabalhando_Com_Data_e_Feriados()
Dim ndata As Date
Dim Rng As Range

datest = "20/03/2022"
'
ndata = DateSerial(Year(datest), Month(datest), 1)
ndata = Application.WorksheetFunction.WorkDay(ndata, -1)

        With Sheets("Feriados").Range("A:A")
            Set Rng = .Find(What:=ndata, _
                            After:=.Cells(.Cells.Count), _
                            LookIn:=xlValues, _
                            LookAt:=xlWhole, _
                            SearchOrder:=xlByRows, _
                            SearchDirection:=xlNext, _
                            MatchCase:=False)
            If Not Rng Is Nothing Then
            ndata = Application.WorksheetFunction.WorkDay(ndata, -1)
            End If
        End With

MsgBox (ndata)

End Sub
